\newtcbox{\parsonbox}[2]{
    enhanced,
    nobeforeafter,
    tcbox raise base,
    boxrule=0.4pt,
    top=0mm,
    bottom=0mm,
    right=0mm,
    left=#1,
    arc=2pt,
    boxsep=2pt,
    before upper={\vphantom{dlg}},
    colframe=black,
    coltext=black,
    colback=lightgray,
    overlay={
        \begin{tcbclipinterior}
        \fill[black]
        (frame.south west)
        rectangle node[
            text=white,
            font=\sffamily\bfseries]
            {#2}
            ([xshift=#1]frame.north west);
        \end{tcbclipinterior}
    }
}
